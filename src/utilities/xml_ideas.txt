Could make a data loader class which automates much of the simple tasks of
loading a data structure from an XmlElement.  Some example function calls are
below.

XmlDataLoader loader;
        
loader.addChildNode(
    "thermodynamics", mp_rrho_model).ifAttEquals("type", "RRHO");

loader.addChildTextValue(
    "formation_enthalpy", &m_hform).withUnits("J/mol");

loader.addChildTextValue("steric_factor", &m_steric);
loader.addChildTextValue("linearity", &m_linearity);

loader.addChildTextValue(
    "rotational_temperature", &m_rotational_t).withUnits("K");

loader.addChildTextValueList(
    "vibrational_temperatures", &m_vibrational_energies).withUnits("K");

loader.load(xml_element);
